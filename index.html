    function clamp(v, min, max) {

      return Math.min(max, Math.max(min, v));

    }

    function toNumber(id) {

      const v = parseFloat(document.getElementById(id).value);

      return isNaN(v) ? 0 : v;

    }

    function fmt2(x) {

      return x.toFixed(2);

    // ===== 初始化顯示 =====

    function updatePointsAndDaily() {

      const level = toNumber("levelInput");

      const totalPoints = level * 4;


      const hungerUsed = toNumber("hungerSkillUsed");

      const eco1 = toNumber("ecoSkill1Used");

      const eco2 = toNumber("ecoSkill2Used");

      const eco3 = toNumber("ecoSkill3Used");


      const used = hungerUsed + eco1 + eco2 + eco3;

      const available = totalPoints - used;


      document.getElementById("usedPointsDisplay").value = used;

      document.getElementById("availablePointsDisplay").value = available;


      const warn = document.getElementById("pointsWarning");

      if (available < 0) {

        warn.style.display = "block";

      } else {

        warn.style.display = "none";

      }


      const hungerMax = toNumber("hungerMaxInput");

      const foodTypeEls = document.querySelectorAll('input[name="foodType"]');

      let foodValue = 10;

      foodTypeEls.forEach(el => { if (el.checked) foodValue = parseFloat(el.value); });

      const dailyHealth = hungerMax * foodValue * 2.4;

      document.getElementById("dailyHealthDisplay").value = fmt2(dailyHealth);

    }

    // ===== 戰鬥計算相關 =====

    function getRankMultiplier(step) {

      // step: 1-120

      const s = clamp(step, 1, 120);

      const smallBonus = s * 0.0025; // 0.25% per step

      const bigSteps = Math.floor((s - 1) / 4);

      const bigBonus = bigSteps * 0.005; // 0.5% per big rank

      const total = smallBonus + bigBonus;

      return 1 + total;

    }


    function getPillMultiplier() {

      const idx = parseInt(document.getElementById("pillSlider").value, 10);

      if (idx === 0) return 0.4;

      if (idx === 1) return 1.0;

      return 1.6;

    }


    function getBuffDebuffMultiplier() {

      const buff = parseFloat(document.getElementById("buffSlider").value) || 0;

      const debuff = parseFloat(document.getElementById("debuffSlider").value) || 0;

      const mBuff = 1 + buff / 100;

      const mDebuff = 1 - debuff / 100;

      return mBuff * mDebuff;

    }


    function getAmmoMultiplier() {

      const els = document.querySelectorAll('input[name="ammoType"]');

      let v = 1.4;

      els.forEach(el => { if (el.checked) v = parseFloat(el.value); });

      return v;

    }


    // 期望傷害

    function computeExpectedDamage(baseAttack, precisionRate, critChanceRate, critDamageMult) {

      // Miss -> 0.5 * A, no crit

      const P = clamp(precisionRate, 0, 1);

      const C = clamp(critChanceRate, 0, 1);

      const A = baseAttack;


      const dmg = A * (0.5 * (1 - P) + P * (1 - C) + P * C * critDamageMult);

      return dmg;

    }


    // 簡化假設：每次攻擊消耗固定體力（讓使用者可以自行再改進）

    const HEALTH_PER_HIT = 10;


    function computeCycleAndDPD(expectedDamage, dailyHealth) {

      if (HEALTH_PER_HIT <= 0) return { cycle: 0, dpd: 0 };

      // 週期：從滿 Health + Hunger 消耗完體力

      // 這裡簡化用 dailyHealth / HEALTH_PER_HIT 作為可攻擊次數 proxy

      const hitsPerDay = dailyHealth / HEALTH_PER_HIT;

      const dpd = hitsPerDay * expectedDamage;

      // 一個「週期」可以理解為把當前 dailyHealth 都用完

      const cycle = hitsPerDay * expectedDamage;

      return { cycle, dpd };

    }


    // ===== 三種流派的加點邏輯（簡化版） =====

    // 所有戰鬥技能上限 10 等，成本 1+2+...+10=55 點；這裡我先假設每等成本 1 點，總可用點數直接分配（之後如要細化可改）

    const STAT_NAMES = ["Attack", "Precision", "CritChance", "CritDamage", "Armor", "Dodge", "Life"];


    function allocatePoints(mode, availablePoints) {

      // mode: "conservative" / "balanced" / "burst"

      // 簡單按比例分配，並限制 0~10 等

      let weights;

      if (mode === "conservative") {

        // 穩健流：重 Precision / Life / Armor

        weights = {

          Attack: 1.0,

          Precision: 1.5,

          CritChance: 0.8,

          CritDamage: 0.7,

          Armor: 1.2,

          Dodge: 1.0,

          Life: 1.3

        };

      } else if (mode === "burst") {

        // 爆發流：重 Attack / CritDamage / CritChance

        weights = {

          Attack: 1.6,

          Precision: 1.0,

          CritChance: 1.3,

          CritDamage: 1.5,

          Armor: 0.6,

          Dodge: 0.7,

          Life: 0.8

        };

      } else {

        // 平衡流

        weights = {

          Attack: 1.3,

          Precision: 1.2,

          CritChance: 1.1,

          CritDamage: 1.1,

          Armor: 0.9,

          Dodge: 0.9,

          Life: 1.0

        };

      }

      const totalWeight = STAT_NAMES.reduce((s, name) => s + weights[name], 0);

      const result = {};

      STAT_NAMES.forEach(name => {

        const alloc = (availablePoints * weights[name]) / totalWeight;

        // 將點數投成等級，最大 10

        result[name] = clamp(Math.floor(alloc), 0, 10);

      });

      // 簡單微調：如果還有剩餘點數，就依照 Attack > Precision > CritDamage > CritChance > Life > Armor > Dodge 的優先順序補到上限 10

      let used = STAT_NAMES.reduce((s, name) => s + result[name], 0);

      let remaining = availablePoints - used;

      const order = ["Attack", "Precision", "CritDamage", "CritChance", "Life", "Armor", "Dodge"];

      let idx = 0;

      while (remaining > 0 && idx < order.length) {

        const key = order[idx];

        if (result[key] < 10) {

          result[key] += 1;

          remaining -= 1;

        } else {

          idx += 1;

        }

      }

      return result;

    }


    function levelToCombatStats(stats) {

      const La = stats.Attack;

      const Lp = stats.Precision;

      const Lcc = stats.CritChance;

      const Lcd = stats.CritDamage;

      // Attack skill

      const A_skill = 100 + 20 * La;

      // Precision

      const P = clamp(0.5 + 0.05 * Lp, 0, 1.4); // 上限 140%，但超過 100% 當作 100% 處理可再調整

      const P_used = Math.min(P, 1.0);

      // Crit Chance

      const C = clamp(0.1 + 0.05 * Lcc, 0, 1.0);

      // Crit Damage（倍率）

      const Dc = 1.0 + 0.2 * Lcd;

      return { A_skill, P_used, C, Dc };

    }


    // ===== 主計算流程 =====

    function runCalculation() {

      ensureResultVisible();

      const warningEl = document.getElementById("resultWarning");

      warningEl.style.display = "none";

      warningEl.textContent = "";


      const availablePoints = parseInt(document.getElementById("availablePointsDisplay").value || "0", 10);

      if (isNaN(availablePoints) || availablePoints <= 0) {

        warningEl.textContent = "可用戰鬥技能點為 0 或無效，請確認等級與已用技能點設定。";

        warningEl.style.display = "block";

      }


      const useAdvanced = document.querySelector('input[name="useAdvanced"]:checked').value;

      const dailyHealth = parseFloat(document.getElementById("dailyHealthDisplay").value) || 0;


      // 乘數部分（進階才用）

      let externalMultiplier = 1.0;

      if (useAdvanced === "yes") {

        const weaponDamage = toNumber("weaponDamage");

        // 這裡簡化：Weapon Damage 視作加到 Attack skill 上的額外固定值

        // Armor / Dodge / Life 對輸出不直接加成，暫不入乘數

        const ammoM = getAmmoMultiplier();

        const pillM = getPillMultiplier();

        const buffDebuffM = getBuffDebuffMultiplier();

        const rankStep = toNumber("rankStep");

        const rankM = getRankMultiplier(rankStep);


        externalMultiplier = ammoM * pillM * buffDebuffM * rankM;


        // weaponDamage 會在 baseAttack 中疊加

        // 其他裝備加成（crit chance, crit dmg, precision, armor, dodge）會加在對應 stat 中

      }


      // 定義三種流派（若 useAdvanced=no，只計算平衡流）

      const flows = useAdvanced === "yes"

        ? [

            { key: "conservative", label: "穩健流", desc: "偏重命中與生存，減少傷害波動。" },

            { key: "balanced", label: "平衡流", desc: "兼顧命中與暴擊，在穩定與爆發之間取平衡。" },

            { key: "burst", label: "爆發流", desc: "偏重攻擊與暴擊倍率，追求最高爆發傷害。" }

          ]

        : [

            { key: "balanced", label: "平衡流", desc: "在可用技能點下給出一組較均衡的戰鬥加點。" }

          ];


      const allocationTbody = document.querySelector("#allocationTable tbody");

      const damageTbody = document.querySelector("#damageTable tbody");

      allocationTbody.innerHTML = "";

      damageTbody.innerHTML = "";


      let bestDpd = -1;

      let bestFlowLabel = "";


      flows.forEach(flow => {

        const alloc = allocatePoints(flow.key, Math.max(availablePoints, 0));


        // 基礎戰鬥技能

        const baseStats = levelToCombatStats(alloc);

        let { A_skill, P_used, C, Dc } = baseStats;


        // 進階裝備加成調整

        if (useAdvanced === "yes") {

          const weaponDamage = toNumber("weaponDamage");

          const weaponCritChance = toNumber("weaponCritChance") / 100;

          const helmetCritDamage = toNumber("helmetCritDamage") / 100;

          const glovesPrecision = toNumber("glovesPrecision") / 100;


          A_skill += weaponDamage;

          C = clamp(C + weaponCritChance, 0, 1);

          Dc = Dc + helmetCritDamage;

          P_used = clamp(P_used + glovesPrecision, 0, 1);

        }


        // 最終攻擊力

        const baseAttack = A_skill * (useAdvanced === "yes" ? externalMultiplier : 1.0);

        const expected = computeExpectedDamage(baseAttack, P_used, C, Dc);

        const cycleD = computeCycleAndDPD(expected, dailyHealth);


        if (cycleD.dpd > bestDpd) {

          bestDpd = cycleD.dpd;

          bestFlowLabel = flow.label;

        }


        // 加點表

        const trAlloc = document.createElement("tr");

        trAlloc.innerHTML = `

          <td><div class="flow-name">${flow.label}</div></td>

          <td><div class="flow-desc">${flow.desc}</div></td>

          <td>${alloc.Attack}</td>

          <td>${alloc.Precision}</td>

          <td>${alloc.CritChance}</td>

          <td>${alloc.CritDamage}</td>

          <td>${alloc.Armor}</td>

          <td>${alloc.Dodge}</td>

          <td>${alloc.Life}</td>

        `;

        allocationTbody.appendChild(trAlloc);


        // 傷害表

        const trDmg = document.createElement("tr");

        trDmg.innerHTML = `

          <td>${flow.label}</td>

          <td>${fmt2(expected)}</td>

          <td>${fmt2(cycleD.cycle)}</td>

          <td>${fmt2(cycleD.dpd)}</td>

        `;

        damageTbody.appendChild(trDmg);

      });


      const topDpdText = document.getElementById("topDpdText");

      if (bestDpd >= 0) {

        topDpdText.textContent = `目前在你的設定下，DPD 最高的是「${bestFlowLabel}」，約為 ${fmt2(bestDpd)} 點傷害／日。`;

      } else {

        topDpdText.textContent = "";

      }

    }
